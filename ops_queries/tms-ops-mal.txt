SELECT
    t_m_s_tenant_profiles.id + 940,
    t_m_s_tenant_profiles.termsofleasecde AS awardnotice_no,
    CASE
        WHEN t_m_s_tenant_profiles.affiliate = 1 THEN 10
        WHEN NOT t_m_s_tenant_profiles.affiliate = 1 THEN NULL
    END AS classification,
    NULL AS classification_update,
    t_m_s_tenant_profiles.turnoverdte AS turnover_date,
    NULL AS turnover_date_update,
    t_m_s_tenant_profiles.modeofpayment AS payment_type,
    NULL AS payment_type_update,
    "" AS store_hour_type,
    NULL AS store_hour_type_update,
    t_m_s_tenant_profiles.commencedte AS opening_date,
    NULL AS opening_date_update,
    "" AS branch_cor,
    "" AS address,
    NULL AS branch_cor_address_update,
    tof.turnoverremarks AS remarks,
    NULL AS remarks_update,
    tof.inspected AS inspected_by,
    NULL AS inspected_by_update,
    tof.noted AS noted_by,
    NULL AS noted_by_update,
    tof.recieved AS conformed_by,
    NULL AS conformed_by_update,
    '1990-01-01 12:00:00' AS created_at,
    '1990-01-01 12:00:00' AS updated_at
FROM ingressfile AS t_m_s_tenant_profiles
JOIN turnoverfile AS tof 
    ON t_m_s_tenant_profiles.ingresscde = tof.referencecde
GROUP BY t_m_s_tenant_profiles.termsofleasecde;


SELECT 
    'ATP' AS permit_type, 
    atpnum AS permit_no,
    dte_expired AS expiration_date,
    ingress.termsofleasecde AS awardnotice_no,
    '10' AS status,
    0 AS created_by,
    0 AS removed_by,
    '1990-01-01 12:00:00' AS removed_at,
    '1990-01-01 12:00:00' AS created_at,
    '1990-01-01 12:00:00' AS updated_at
FROM ingressotherfile AS t_m_s_tenant_profile_x_permits
LEFT JOIN ingressfile AS ingress 
    ON t_m_s_tenant_profile_x_permits.ingresscde = ingress.ingresscde
WHERE atpnum IS NOT NULL
  AND TRIM(atpnum) <> ''
  AND UPPER(TRIM(atpnum)) NOT IN ('N/A', 'NA', 'NONE', 'NULL')

UNION ALL

SELECT
    t_m_s_meters.id + 1151 AS id,
    1 AS mall_id,
    CONCAT('MTR', LPAD(id + 1151, 5, '0')) AS meter_code,
    t_m_s_meters.meternum AS meter_no,
    t_m_s_meters.multiplier,
    CASE
        WHEN t_m_s_meters.meterctype = 'CHILLED WATER' THEN 'CHILLED-WATER'
        WHEN t_m_s_meters.meterctype = 'ELECTRICITY - HYPO' THEN 'ELECTRICITY-HYPO'
        ELSE t_m_s_meters.meterctype
    END AS charge_type,
    1 AS created_by,
    1 AS updated_by,
    CASE
        WHEN t_m_s_meters.deleted = 0
        AND t_m_s_meters.stat = 'Active'
        AND t_m_s_meters.termsofleasecde IS NOT NULL
        AND t_m_s_meters.termsofleasecde <> '' THEN 10
        WHEN NOT t_m_s_meters.deleted = 0
        OR t_m_s_meters.stat != 'Active'
        OR t_m_s_meters.termsofleasecde IS NULL
        OR t_m_s_meters.termsofleasecde = '' THEN 20
    END AS `status`,
    '1990-01-01 12:00:00' AS created_at,
    '1990-01-01 12:00:00' AS updated_at
FROM
    meterfile AS t_m_s_meters;
