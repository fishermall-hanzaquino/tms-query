SELECT
    t_m_s_award_notices.id,
    t_m_s_award_notices.id AS counter,
    MD5(t_m_s_award_notices.id) AS hashid,
    t_m_s_award_notices.termsofleasecde AS anno,
    2 AS mallid,
    t_m_s_award_notices.offerdte AS date,
    (
        SELECT
            locationfile.id
        FROM
            locationfile
        WHERE
            t_m_s_award_notices.loccde = locationfile.loccde
        LIMIT
            1
    ) AS locationid,
    COALESCE(
        (
            SELECT
                amendparticularsfile.valueto
            FROM
                amendmentfile
                LEFT JOIN amendparticularsfile ON amendmentfile.amendmentcde = amendparticularsfile.amendmentcde
            WHERE
                t_m_s_award_notices.termsofleasecde = amendmentfile.termsofleasecde
                AND amendmentfile.posted = 1
                AND amendparticularsfile.amendfield = 'Store Name'
                AND amendmentfile.effectivedte < '2026-02-28'
            ORDER BY
                amendmentfile.effectivedte DESC
            LIMIT
                1
        ), t_m_s_award_notices.storenme
    ) AS establishmentname,
    (
        SELECT
            tenantaccountfile.id
        FROM
            tenantaccountfile
        WHERE
            tenantaccountfile.clientacctcde = t_m_s_award_notices.clientacctcde
        LIMIT
            1
    ) AS clientid,
    (
        SELECT
            tenanttypefile.id
        FROM
            tenanttypefile
        WHERE
            tenanttypefile.tenantdesc = t_m_s_award_notices.tenanttype
        LIMIT
            1
    ) AS tenanttypeid,
    COALESCE(
        (
            SELECT
                amendparticularsfile.valueto
            FROM
                amendmentfile
                LEFT JOIN amendparticularsfile ON amendmentfile.amendmentcde = amendparticularsfile.amendmentcde
            WHERE
                t_m_s_award_notices.termsofleasecde = amendmentfile.termsofleasecde
                AND amendmentfile.posted = 1
                AND amendparticularsfile.amendfield = 'Corporate Name'
                AND amendmentfile.effectivedte < '2026-02-28'
            ORDER BY
                amendmentfile.effectivedte DESC
            LIMIT
                1
        ), t_m_s_award_notices.clientnme
    ) AS tradename,
    (
        SELECT
            businesscategoryfile.id
        FROM
            businesscategoryfile
        WHERE
            REPLACE(
                REPLACE(businesscategoryfile.busscategorydsc, '\r', ''),
                '\n',
                ''
            ) = REPLACE(
                REPLACE(t_m_s_award_notices.busscategory, '\r', ''),
                '\n',
                ''
            )
        LIMIT
            1
    ) AS categoryid,
    t_m_s_award_notices.permitusage AS permittedusage,
    COALESCE(
        t_m_s_award_notices.tin,
        (
            SELECT
                tenantaccountfile.tenantaccttin
            FROM
                tenantaccountfile
            WHERE
                tenantaccountfile.clientacctcde = t_m_s_award_notices.clientacctcde
            LIMIT
                1
        )
    ) AS tin,
    t_m_s_award_notices.foperiodfrdte AS fitoutfrom,
    t_m_s_award_notices.foperiodtodte AS fitoutto,
    (
        SELECT
            tenantaccountfile.tenantaccttin
        FROM
            tenantaccountfile
        WHERE
            tenantaccountfile.clientacctcde = t_m_s_award_notices.clientacctcde
        LIMIT
            1
    ) AS headofficetin,
    (
        SELECT
            tenantaccountfile.tenantacctaddr
        FROM
            tenantaccountfile
        WHERE
            tenantaccountfile.clientacctcde = t_m_s_award_notices.clientacctcde
        LIMIT
            1
    ) AS headofficeaddress,
    10 AS STATUS,
    CASE
        WHEN t_m_s_award_notices.posted = 0 THEN 10
        WHEN t_m_s_award_notices.posted = 1
        AND NOT contractcde = ''
        AND termsofleasecde IS NOT NULL
        AND termsofleasecde <> '' THEN 20
        WHEN t_m_s_award_notices.terminate = 1 THEN 30
        ELSE 30
    END AS `state`,
    '' AS postedat,
    '1' AS postedby,
    '' AS terminatedby,
    t_m_s_award_notices.terminatedte AS terminatedat,
    t_m_s_award_notices.preparedby AS createdby,
    t_m_s_award_notices.preparedby AS updatedby,
    '1990-01-01 12:00:00' AS created_at,
    '1990-01-01 12:00:00' AS updated_at
FROM
    clientofferfile AS t_m_s_award_notices;



SELECT
    id,
    id AS contracts_tmsawardnoticeid,
    '2' AS contracts_mallid,
    MD5(id) AS contracts_hashid,
    termsofleasecde AS contracts_anno,
    1 AS contracts_counter,
    contractcde AS contracts_contractid,
    10 AS contracts_status,
    CASE
        WHEN leasepostdte IS NULL THEN 10
        WHEN leasepostdte IS NOT NULL
        AND contractcde <> ''
        AND termsofleasecde IS NOT NULL
        AND termsofleasecde <> ''
        AND cancelled != 1
        AND CURRENT_DATE BETWEEN leasetermfrdte AND leasetermtodte
        AND terminatedte IS NULL THEN 20
        WHEN terminatedte IS NOT NULL THEN 30
        ELSE 10
    END AS contracts_state,
    '1990-01-01 12:00:00' AS created_at,
    '1990-01-01 12:00:00' AS updated_at
FROM
    clientofferfile AS t_m_s_contracts
UNION
ALL
SELECT
    id + 1047 AS id,
    id AS contracts_tmsawardnoticeid,
    '2' AS contracts_mallid,
    MD5(id + 1047) AS contracts_hashid,
    termsofleasecde AS contracts_anno,
    1 AS contracts_counter,
    contractrncde AS contracts_contractid,
    10 AS contracts_status,
    CASE
        WHEN postdte IS NULL THEN 10
        WHEN postdte IS NOT NULL
        AND contractrncde <> ''
        AND termsofleasecde IS NOT NULL
        AND termsofleasecde <> ''
        AND cancelled != 1
        AND CURRENT_DATE BETWEEN leasetermfrdte AND leasetermtodte
        AND terminatedte IS NULL THEN 20
        WHEN terminatedte IS NOT NULL THEN 30
        ELSE 10
    END AS contracts_state,
    '1990-01-01 12:00:00' AS created_at,
    '1990-01-01 12:00:00' AS updated_at
FROM
    contractrenewalfile AS t_m_s_contracts;

SELECT
    t_m_s_a_n_rental_charges.id,
    CASE
        WHEN t_m_s_a_n_rental_charges.contract_renew = 1 THEN (
            SELECT
                id + 1047
            FROM
                contractrenewalfile
            WHERE
                t_m_s_a_n_rental_charges.contractrncde = contractrenewalfile.contractrncde
            ORDER BY
                id DESC
            LIMIT
                1
        )
        ELSE clientofferfile.id
    END AS tms_contract_id,
    clientofferfile.id AS awardnoticeid,
    CASE
        WHEN t_m_s_a_n_rental_charges.percentrent = 0 THEN 2
        WHEN t_m_s_a_n_rental_charges.percentrent > 0 THEN 4
    END AS rentalschemeid,
    periodfrdte AS fromdate,
    periodtodte AS todate,
    basicrate AS basicorfixedrate,
    percentrent AS percentagerate,
    '0' AS ari,
    minrent AS minimumrate,
    CASE
        WHEN uom = "Per unit" THEN "PerUnit"
        ELSE "SQM"
    END AS uom,
    secdepchargerate AS secdep,
    secdepmonths AS sdma,
    advrentchargerate AS advrent,
    constructionchargerate AS conbond,
    CASE
        WHEN t_m_s_a_n_rental_charges.cancelled = 0 THEN 10
        ELSE 20
    END AS `status`,
    '0' AS createdby,
    '0' AS updatedby,
    '1990-01-01 12:00:00' AS created_at,
    '1990-01-01 12:00:00' AS updated_at
FROM
    rentalchargesfile AS t_m_s_a_n_rental_charges
    LEFT JOIN clientofferfile ON t_m_s_a_n_rental_charges.transcde = clientofferfile.offercde
    AND t_m_s_a_n_rental_charges.termsofleasecde = clientofferfile.termsofleasecde
ORDER BY
    t_m_s_a_n_rental_charges.id;

SELECT
    t_m_s_a_n_other_charges.id,
    CASE
        WHEN t_m_s_a_n_other_charges.contract_renew = 1 THEN (
            SELECT
                id + 1047
            FROM
                contractrenewalfile
            WHERE
                t_m_s_a_n_other_charges.contractrncde = contractrenewalfile.contractrncde
            ORDER BY
                id DESC
            LIMIT
                1
        )
        ELSE clientofferfile.id
    END AS tms_contract_id,
    clientofferfile.id AS tms_awardnotice_id,
    0 AS applicable_charge_id,
    t_m_s_a_n_other_charges.rate,
    CASE
        WHEN t_m_s_a_n_other_charges.costtypeuom = 'Per sqm.' OR t_m_s_a_n_other_charges.costtypeuom = 'Per sqm' THEN 'sqm'
        WHEN t_m_s_a_n_other_charges.costtypeuom = 'Per unit' THEN 'perconsumption'
    END AS uom,
    t_m_s_a_n_other_charges.othrchadesc as remarks,
    CASE
        WHEN t_m_s_a_n_other_charges.cancelled = 0 THEN 10
        ELSE 20
    END AS `status`,
    '0' AS created_by,
    '0' AS updated_by,
    '1990-01-01 12:00:00' AS created_at,
    '1990-01-01 12:00:00' AS updated_at
FROM
    otherchargesfile AS t_m_s_a_n_other_charges
    LEFT JOIN clientofferfile ON t_m_s_a_n_other_charges.transcde = clientofferfile.offercde
    AND t_m_s_a_n_other_charges.termsofleasecde = clientofferfile.termsofleasecde
WHERE
    t_m_s_a_n_other_charges.othrchadesc NOT IN (
        'SECURITY DEPOSIT',
        'ADVANCE RENT',
        'CONSTRUCTION BOND'
    )
    AND CURDATE() BETWEEN effectfrdte AND effecttodte
ORDER BY
    t_m_s_a_n_other_charges.id;
