// epf

SELECT
    e_b_m_events_epf_information.id,
    e_b_m_events_epf_information.eventcde AS epf_id,
    e_b_m_events_epf_information.dte_received AS loi,
    CASE
        WHEN e_b_m_events_epf_information.branchcde = 'BR000001' THEN 2
        ELSE 1
    END AS branch, 
    CASE
        WHEN e_b_m_events_epf_information.clienttype_id = 1 THEN 1
        WHEN e_b_m_events_epf_information.clienttype_id = 2 THEN 2
        ELSE 1
    END AS client_type,
    CASE
        WHEN e_b_m_events_epf_information.clientoffer_id IS NOT NULL THEN e_b_m_events_epf_information.clientoffer_id
        ELSE e_b_m_events_epf_information.org_id
    END AS client_id,
    e_b_m_events_epf_information.epftype_id AS event_type,
    eventclassfile.eventclass as classification,
    e_b_m_events_epf_information.event_type AS transaction_type,
    e_b_m_events_epf_information.title,
    e_b_m_events_epf_information.location,
    e_b_m_events_epf_information.adv_rent,
    e_b_m_events_epf_information.sec_dep,
    e_b_m_events_epf_information.request AS requests,
    e_b_m_events_epf_information.remarks,
    e_b_m_events_epf_information.recommendation AS recommendations,
    CASE
        WHEN e_b_m_events_epf_information.cancelled = 1 THEN 20
        WHEN moa.id IS NOT NULL THEN 30
        ELSE 10
    END AS 'status'
FROM eventfile as e_b_m_events_epf_information
LEFT JOIN eventclassfile
    ON e_b_m_events_epf_information.eventclass_id = eventclassfile.id
LEFT JOIN moafile AS moa
    ON e_b_m_events_epf_information.id = moa.epf_id;

// rental cost
SELECT
    e_b_m_events_rental_costs.id,
    CASE
        WHEN e_b_m_events_rental_costs.event_id = 0 AND moa.epf_id <> 0 THEN moa.epf_id
        ELSE e_b_m_events_rental_costs.event_id
    END AS epf_id,
    e_b_m_events_rental_costs.moa_id,
    CASE
        WHEN e_b_m_events_rental_costs.bill_term = 'Per Day' THEN 1
        WHEN e_b_m_events_rental_costs.bill_term = 'Per Week' THEN 2
        WHEN e_b_m_events_rental_costs.bill_term = 'Per Month' THEN 3
        WHEN e_b_m_events_rental_costs.bill_term = 'Per Year' THEN 4
        ELSE 5
    END AS billing_type,
    CASE
        WHEN e_b_m_events_rental_costs.percent > 0 THEN 2
        WHEN e_b_m_events_rental_costs.rate > 0 THEN 1
        WHEN e_b_m_events_rental_costs.rate = 0 AND percent = 0 THEN 3
        ELSE 3
    END AS rental_scheme,
    e_b_m_events_rental_costs.dte_from AS date_from,
    e_b_m_events_rental_costs.dte_to AS date_to,
    e_b_m_events_rental_costs.rate AS fixed_rate,
    e_b_m_events_rental_costs.percent AS percentage,
    e_b_m_events_rental_costs.min_rate,
    e_b_m_events_rental_costs.remarks
FROM eventperiodcoveredfile AS e_b_m_events_rental_costs
LEFT JOIN moafile AS moa
    ON e_b_m_events_rental_costs.moa_id = moa.id;

// moa
SELECT
    e_b_m_events_moa_information.id,
    e_b_m_events_moa_information.epf_id,
    e_b_m_events_moa_information.moacde AS moa_id,
    epf.dte_received AS loi,
    CASE
        WHEN e_b_m_events_moa_information.branchcde = 'BR000001' THEN 2
        ELSE 1
    END AS branch,
    CASE
        WHEN epf.clienttype_id = 1 THEN 1
        WHEN epf.clienttype_id = 2 THEN 2
        ELSE 1
    END AS client_type,
    CASE
        WHEN epf.clientoffer_id IS NOT NULL THEN epf.clientoffer_id
        ELSE epf.org_id
    END AS client_id,
    epf.epftype_id AS event_type,
    eventclassfile.eventclass as classification,
    epf.event_type AS transaction_type,
    epf.title,
    epf.location,
    epf.adv_rent,
    epf.sec_dep,
    epf.request AS requests,
    epf.remarks,
    epf.recommendation AS recommendations,
    e_b_m_events_moa_information.floor AS floor_area,
    e_b_m_events_moa_information.caterer AS caterer,
    e_b_m_events_moa_information.adv_rent_add AS adv_rent,
    CASE
        WHEN e_b_m_events_moa_information.roll_over <> '' OR e_b_m_events_moa_information.roll_over IS NOT NULL THEN e_b_m_events_moa_information.roll_over
        ELSE NULL
    END AS sec_dep_moa_id,
    CASE
        WHEN e_b_m_events_moa_information.sec_dep_add <> 0 THEN e_b_m_events_moa_information.roll_over_amt
        ELSE e_b_m_events_moa_information.roll_over_amt
    END AS sec_dep,
    // to do and check - rethink logic
    CASE
        WHEN e_b_m_events_moa_information.roll_over = '' OR e_b_m_events_moa_information.roll_over IS NULL THEN 10
        ELSE 20
    END AS sec_dep_status,
    // to do check status
    CASE
        WHEN e_b_m_events_moa_information.cancelled = 1 THEN 20
        WHEN e_b_m_events_moa_information.approved_by > 0 THEN 40
        ELSE 30
    END AS 'status'
    //e_b_m_events_moa_information.due_date AS duedate
    // e_b_m_events_moa_information.
FROM moafile AS e_b_m_events_moa_information
LEFT JOIN eventfile AS epf
    ON e_b_m_events_moa_information.epf_id = epf.id
LEFT JOIN eventclassfile
    ON epf.eventclass_id = eventclassfile.id;
